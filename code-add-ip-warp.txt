#!/bin/bash

ip_file="/root/ipopenwarp.txt"

# Kiểm tra xem tệp IP có tồn tại không
if [ -e "$ip_file" ]; then
  # Đọc danh sách IP từ tệp
  while IFS= read -r ip; do
    # Kiểm tra nếu IP không rỗng
    if [ -n "$ip" ]; then
      # Sử dụng warp-cli để thêm IP vào excluded routes
      warp-cli add-excluded-route "$ip"
      echo "Added $ip to excluded routes"
    fi
  done < "$ip_file"
else
  echo "File $ip_file does not exist."
fi
